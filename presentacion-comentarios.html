    transform: scale(1.2);
}
JavaScript 
let likes = 0;
let isLiked = false;

function toggleLike() {
    const btn = document.getElementById('likeBtn');
    
    if (isLiked) {
        likes--;  // Quitar like
        btn.classList.remove('liked');
    } else {
        likes++;  // Agregar like
        btn.classList.add('liked');
    }
    
    isLiked = !isLiked;  // Cambiar estado
    document.getElementById('likeCount').textContent = likes;
}

Calificación con Estrellas - Parte 2

JavaScript 
function rate(stars) {
    // Seleccionar todas las estrellas
    const allStars = document.querySelectorAll('#stars .star');
    
    // Recorrer cada estrella
    allStars.forEach((star, index) => {
        if (index < stars) {
            // Agregar clase 'active' a las primeras N estrellas
            star.classList.add('active');
        } else {
            // Quitar clase 'active' del resto
            star.classList.remove('active');
        }
    });
    
    console.log('Calificación:', stars);
}
Truco: Usa index < stars para activar solo las primeras N estrellas

Estrellas con Hover - CSS Avanzado

Para iluminar estrellas al pasar el mouse:

CSS con selector ~ 
/* El selector ~ selecciona TODOS los hermanos siguientes */

.star:hover {
    color: #ffd700;  /* Estrella donde está el mouse */
}

.star:hover ~ .star {
    color: #ddd;     /* Estrellas DESPUÉS del mouse */
}
¿Cómo funciona?
Si pasas el mouse sobre la estrella 3:
- Estrella 1, 2, 3 → doradas
- Estrella 4, 5 → grises (hermanos siguientes)

Agregar Comentario - JavaScript

JavaScript 
let selectedRating = 0;  // Calificación seleccionada
let comments = [];       // Array de comentarios

function addComment() {
    // Obtener valores
    const text = document.getElementById('commentText').value;
    const user = document.getElementById('currentUser').textContent;
    
    // Validar
    if (!text || selectedRating === 0) {
        alert('Completa todos los campos');
        return;
    }
    
    // Crear nuevo comentario
    const newComment = {
        product: 50741,
        score: selectedRating,
        description: text,
        user: user,
        dateTime: new Date().toISOString().slice(0, 19).replace('T', ' ')
    };
    
    // Agregar al inicio del array
    comments.unshift(newComment);
    
    // Actualizar vista
    showComments(comments);
    
    // Limpiar formulario
    document.getElementById('commentText').value = '';
    selectedRating = 0;
}
