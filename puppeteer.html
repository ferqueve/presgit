const puppeteer = require('puppeteer');

class TestSuite {
  constructor() {
    this.results = [];
  }
  
  async runTest(name, testFunction) {
    try {
      console.log(`üß™ Ejecutando: ${name}`);
      await testFunction();
      this.results.push({name, status: 'PASS'});
      console.log(`‚úÖ ${name}: PASSED`);
    } catch (error) {
      this.results.push({name, status: 'FAIL', error: error.message});
      console.log(`‚ùå ${name}: FAILED - ${error.message}`);
    }
  }
  
  async runAllTests() {
    const browser = await puppeteer.launch();
    const page = await browser.newPage();
    
    await this.runTest('Homepage loads', async () => {
      await page.goto('https://mysite.com');
      await page.waitForSelector('h1');
    });
    
    await this.runTest('Contact form works', async () => {
      await page.goto('https://mysite.com/contact');
      await page.type('#name', 'Test User');
      await page.type('#email', 'test@test.com');
      await page.click('#submit');
      await page.waitForSelector('.success');
    });
    
    await browser.close();
    this.generateReport();
  }
  
  generateReport() {
    const passed = this.results.filter(r => r.status === 'PASS').length;
    const total = this.results.length;
    console.log(`\nüìä Resumen: ${passed}/${total} tests pasaron`);
  }
}

new TestSuite().runAllTests();
