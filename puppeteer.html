Instalación y Configuración

Paso 1: Instalar Node.js

Asegúrate de tener Node.js instalado (versión 12 o superior)

Terminal 
node --version
npm --version

Paso 2: Crear Proyecto

Inicializa un nuevo proyecto de Node.js

Terminal 
mkdir mi-proyecto-puppeteer
cd mi-proyecto-puppeteer
npm init -y

Paso 3: Instalar Puppeteer

Instala Puppeteer - esto también descargará Chromium automáticamente

Terminal 
# Instalación básica
npm install puppeteer

# O si prefieres no descargar Chromium
npm install puppeteer-core
¡Importante! La instalación de Puppeteer descarga una versión completa de Chromium (~170MB). Si tienes problemas de espacio, puedes usar puppeteer-core y configurar tu propio navegador.
Prueba la Instalación

Crea un archivo test.js y ejecuta este código:

console.log('Versión:', puppeteer.version);

Tu Primer Script con Puppeteer

Captura de Pantalla Básica

ejemplo-basico.js 
const puppeteer = require('puppeteer');

(async () => {
  // 1. Lanzar el navegador
  const browser = await puppeteer.launch();
  
  // 2. Crear una nueva página
  const page = await browser.newPage();
  
  // 3. Navegar a una URL
  await page.goto('https://example.com');
  
  // 4. Tomar captura de pantalla
  await page.screenshot({ 
    path: 'ejemplo.png' 
  });
  
  // 5. Cerrar el navegador
  await browser.close();
  
  console.log('¡Captura guardada como ejemplo.png!');
})();

Interacción con Páginas Web

Seleccionar Elementos

seleccionar-elementos.js 
const puppeteer = require('puppeteer');

(async () => {
  const browser = await puppeteer.launch();
  const page = await browser.newPage();
  
  await page.goto('https://example.com');
  
  // Seleccionar por selector CSS
  const titulo = await page.$('h1');
  
  // Obtener texto
  const textoTitulo = await page.$eval(
    'h1', 
    el => el.textContent
  );
  
  console.log('Título:', textoTitulo);
  
  await browser.close();
})();

Hacer Clic y Escribir

interaccion.js 
const puppeteer = require('puppeteer');

(async () => {
  const browser = await puppeteer.launch({
    headless: false // Ver el navegador
  });
  const page = await browser.newPage();
  
  await page.goto('https://google.com');
  
  // Escribir en campo de búsqueda
  await page.type('input[name="q"]', 'Puppeteer');
  
  // Hacer clic en botón buscar
  await page.click('input[value="Buscar con Google"]');
  
  // Esperar a que cargue
  await page.waitForNavigation();
  
  console.log('Búsqueda realizada!');
  
  await browser.close();
})();

Esperas y Sincronización

¡Muy Importante! Las páginas web cargan de forma asíncrona. Debes esperar a que los elementos estén disponibles antes de interactuar con ellos.

Tipos de Espera

esperas.js 
const puppeteer = require('puppeteer');

(async () => {
  const browser = await puppeteer.launch();
  const page = await browser.newPage();
  
  await page.goto('https://example.com');
  
  // 1. Esperar por tiempo fijo
  await page.waitForTimeout(2000); // 2 segundos
  
  // 2. Esperar a que aparezca un elemento
  await page.waitForSelector('h1');
  
  // 3. Esperar navegación
  await page.click('a[href="/about"]');
  await page.waitForNavigation();
  
  // 4. Esperar por función personalizada
  await page.waitForFunction(() => {
    return document.title.includes('About');
  });
  
  await browser.close();
})();

Estrategias de Espera

waitForTimeout()

Espera fija - úsala solo cuando sea necesario

await page.waitForTimeout(1000);
waitForSelector()

Espera hasta que aparezca un elemento

await page.waitForSelector('.button');
waitForNavigation()

Espera a que termine la navegación

await page.waitForNavigation();
waitForFunction()

Espera hasta que una condición sea verdadera

await page.waitForFunction(...);

Ejemplo Práctico: Formulario Dinámico

formulario-dinamico.js 
const puppeteer = require('puppeteer');

(async () => {
  const browser = await puppeteer.launch({ headless: false });
  const page = await browser.newPage();
  
  await page.goto('https://ejemplo-formulario.com');
  
  // Llenar formulario paso a paso
  await page.type('#nombre', 'Juan Pérez');
  
  // Esperar a que aparezca el campo email (puede cargarse dinámicamente)
  await page.waitForSelector('#email', { visible: true });
  await page.type('#email', 'juan@example.com');
  
  // Clic en enviar y esperar respuesta
  await Promise.all([
    page.waitForNavigation({ waitUntil: 'networkidle0' }),
    page.click('#enviar')
  ]);
  
  // Verificar que llegamos a la página de confirmación
  await page.waitForSelector('.mensaje-exito');
  
  console.log('Formulario enviado exitosamente!');
  
  await browser.close();
})();
Mejores Prácticas
  • Siempre usa waitForSelector() antes de interactuar con elementos
  • Evita waitForTimeout() - es impredecible
  • Usa { waitUntil: 'networkidle0' } para páginas con muchas solicitudes AJAX
  • Combina acciones con esperas usando Promise.all()

Testing Automatizado con Puppeteer

Ejemplo de Test E2E Completo

test-login.js 
const puppeteer = require('puppeteer');

async function testLogin() {
  const browser = await puppeteer.launch({ headless: false });
  const page = await browser.newPage();
  
  try {
    console.log('🚀 Iniciando test de login...');
    
    // 1. Navegar a la página de login
    await page.goto('https://example.com/login');
    await page.waitForSelector('#username');
    
    // 2. Llenar credenciales
    await page.type('#username', 'usuario@test.com');
    await page.type('#password', 'password123');
    
    // 3. Hacer clic en iniciar sesión
    await Promise.all([
      page.waitForNavigation(),
      page.click('#login-btn')
    ]);
    
    // 4. Verificar que el login fue exitoso
    const welcomeMessage = await page.$('.welcome-message');
    
    if (welcomeMessage) {
      console.log('✅ Test PASÓ: Login exitoso');
      
      // Verificar elementos de la página de inicio
      const profileLink = await page.$('#profile-link');
      const logoutBtn = await page.$('#logout-btn');
      
      if (profileLink && logoutBtn) {
        console.log('✅ Test PASÓ: Elementos de navegación presentes');
      } else {
        console.log('❌ Test FALLÓ: Faltan elementos de navegación');
      }
      
    } else {
      console.log('❌ Test FALLÓ: Login no exitoso');
    }
    
  } catch (error) {
    console.log('❌ Test ERROR:', error.message);
  } finally {
    await browser.close();
  }
}

testLogin();

Proyecto 1: Monitor de Precios

Crea un script que monitoree precios de productos y te notifique cuando bajen.

const puppeteer = require('puppeteer');

async function monitorPrice(url, selector) {
  const browser = await puppeteer.launch();
  const page = await browser.newPage();
  
  await page.goto(url);
  
  const price = await page.$eval(selector, 
    el => el.textContent.replace(/[^\d.,]/g, '')
  );
  
  console.log(`Precio actual: $${price}`);
  
  // Aquí puedes agregar lógica para:
  // - Guardar precio en base de datos
  // - Enviar notificación si baja
  // - Generar gráficos históricos
  
  await browser.close();
  return parseFloat(price);
}

// Ejecutar cada hora
setInterval(() => {
  monitorPrice('https://tienda.com/producto', '.price');
}, 3600000);

Proyecto 2: Generador de Reportes

Automatiza la generación de reportes PDF de dashboards web.

const puppeteer = require('puppeteer');

async function generateReport(url) {
  const browser = await puppeteer.launch();
  const page = await browser.newPage();
  
  // Configurar viewport para reportes
  await page.setViewport({ 
    width: 1200, 
    height: 800 
  });
  
  await page.goto(url, { 
    waitUntil: 'networkidle0' 
  });
  
  // Esperar a que carguen los gráficos
  await page.waitForSelector('.chart-container');
  
  // Generar PDF
  await page.pdf({
    path: `reporte-${Date.now()}.pdf`,
    format: 'A4',
    printBackground: true,
    margin: { top: '20px', bottom: '20px' }
  });
  
  await browser.close();
  console.log('Reporte generado exitosamente!');
}

generateReport('https://dashboard.com');

Proyecto 3: Bot de Redes Sociales

Automatiza publicaciones y interacciones básicas (¡con responsabilidad!).

const puppeteer = require('puppeteer');

async function postContent(message) {
  const browser = await puppeteer.launch({ 
    headless: false 
  });
  const page = await browser.newPage();
  
  // Login (debes manejar autenticación)
  await page.goto('https://red-social.com/login');
  
  // Esperar y llenar credenciales
  await page.waitForSelector('#username');
  await page.type('#username', process.env.USERNAME);
  await page.type('#password', process.env.PASSWORD);
  
  await Promise.all([
    page.waitForNavigation(),
    page.click('#login-btn')
  ]);
  
  // Crear nueva publicación
  await page.waitForSelector('#new-post');
  await page.click('#new-post');
  
  await page.type('.post-textarea', message);
  await page.click('.publish-btn');
  
  console.log('Publicación creada!');
  await browser.close();
}

// Usar con moderación y respeto
postContent('¡Hola desde Puppeteer!');

Proyecto 4: Testing Suite

Crea una suite completa de tests automatizados para tu sitio web.

const puppeteer = require('puppeteer');

class TestSuite {
  constructor() {
    this.results = [];
  }
  
  async runTest(name, testFunction) {
    try {
      console.log(`🧪 Ejecutando: ${name}`);
      await testFunction();
      this.results.push({name, status: 'PASS'});
      console.log(`✅ ${name}: PASSED`);
    } catch (error) {
      this.results.push({name, status: 'FAIL', error: error.message});
      console.log(`❌ ${name}: FAILED - ${error.message}`);
    }
  }
  
  async runAllTests() {
    const browser = await puppeteer.launch();
    const page = await browser.newPage();
    
    await this.runTest('Homepage loads', async () => {
      await page.goto('https://mysite.com');
      await page.waitForSelector('h1');
    });
    
    await this.runTest('Contact form works', async () => {
      await page.goto('https://mysite.com/contact');
      await page.type('#name', 'Test User');
      await page.type('#email', 'test@test.com');
      await page.click('#submit');
      await page.waitForSelector('.success');
    });
    
    await browser.close();
    this.generateReport();
  }
  
  generateReport() {
    const passed = this.results.filter(r => r.status === 'PASS').length;
    const total = this.results.length;
    console.log(`\n📊 Resumen: ${passed}/${total} tests pasaron`);
  }
}

new TestSuite().runAllTests();
